function GithubAuth() {
 
  var opts = {
    version: '3.0.0',
    timeout: 30000
  };
 
  var proxy = bag.proxy();
  if (proxy) {
    proxy = url.parse(proxy);
    opts.proxy = {
      host: proxy.hostname,
      port: proxy.port
    };
  }
 
  this.gh = new github(opts);
}

};
